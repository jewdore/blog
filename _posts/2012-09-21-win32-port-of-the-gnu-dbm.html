在我的VC++2008, 需要对getopt.c做如下修改， 才能通过编译

1 #ifdef    __GNU_LIBRARY__2 #undef    alloca3 /* Don't include stdlib.h for non-GNU C libraries because some of them4    contain conflicting prototypes for getopt.  */5 #include <stdlib.h>6 #else    /* Not GNU C library.  */7 //#define    __alloca    alloca //注释掉这行, alloca在stack上分配内存8 #define    __alloca    malloc  //改成malloc, malloc在heap上分配内存9 #endif    /* GNU C library.  */
